mysql> WITH RECORDS(Ename, Hire_date, Six_after) AS (
    ->     SELECT Ename,
    ->            Hire_date,
    ->            DATE_ADD(Hire_date, INTERVAL 6 MONTH)
    ->                AS Six_after
    ->     FROM EMPLOYEE
    -> )
    -> SELECT RECORDS.Ename,
    ->        RECORDS.Hire_date,
    ->        DATE_ADD(
    ->                RECORDS.Six_after,
    ->                INTERVAL
    ->                IF(
    ->                        0 = WEEKDAY(RECORDS.Six_after),
    ->                        0 - WEEKDAY(RECORDS.Six_after),
    ->                        7 - WEEKDAY(RECORDS.Six_after) + 0
    ->                    )
    ->                DAY
    ->            ) AS Salary_review_date
